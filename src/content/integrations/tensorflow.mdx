@step

 &nbsp; &nbsp;"""Step that trains a tiny model using the input dataset."""

 &nbsp; &nbsp; &nbsp; &nbsp;tf.keras.layers.Dense(10, activation='relu', input_shape=(5,)),

 &nbsp; &nbsp;])

 &nbsp; &nbsp;model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

 &nbsp; &nbsp;# Train for just one epoch
 &nbsp; &nbsp;model.fit(dataset, epochs=1)

 &nbsp; &nbsp;return model

@pipeline(enable_cache=False)
def tiny_model_pipeline():
 &nbsp; &nbsp;dataset = load_dataset()

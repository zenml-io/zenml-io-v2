import mlflow
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score

mlflow.set_tracking_uri("http://localhost:5000")
mlflow.set_experiment("my_experiment")

with mlflow.start_run():
 &nbsp; &nbsp;# Load and preprocess data
 &nbsp; &nbsp;X, y = load_data()
 &nbsp; &nbsp;X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

 &nbsp; &nbsp;# Train model
 &nbsp; &nbsp;model = RandomForestClassifier()
 &nbsp; &nbsp;model.fit(X_train, y_train)

 &nbsp; &nbsp;# Log parameters
 &nbsp; &nbsp;mlflow.log_param("n_estimators", model.n_estimators)
 &nbsp; &nbsp;mlflow.log_param("max_depth", model.max_depth)

 &nbsp; &nbsp;# Log metrics
 &nbsp; &nbsp;y_pred = model.predict(X_test)
 &nbsp; &nbsp;accuracy = accuracy_score(y_test, y_pred)
 &nbsp; &nbsp;mlflow.log_metric("accuracy", accuracy)

 &nbsp; &nbsp;# Log model
 &nbsp; &nbsp;mlflow.sklearn.log_model(model, "random_forest_model")

# Retrieve and print the run ID
current_run = mlflow.active_run()
